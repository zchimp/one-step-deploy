```java
public class ArrayReverse {
    public static void reverseArray(int[] arr) {
        if (arr == null || arr.length <= 1) {
            return; // 空数组或单元素数组无需反转
        }
        int left = 0;
        int right = arr.length - 1;
        while (left < right) {
            // 交换左右元素
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            left++;
            right--;
        }
    }

    public static void main(String[] args) {
        int[] arr = {1,2,3,4,5};
        reverseArray(arr);
        System.out.println(Arrays.toString(arr)); // [5,4,3,2,1]
    }
}
```

2. 数组去重（有序数组）
```java
public class ArrayDeduplication {
    public static int removeDuplicates(int[] nums) {
        if (nums.length == 0) return 0;
        int slow = 0; // 慢指针：记录不重复元素的位置
        for (int fast = 1; fast < nums.length; fast++) { // 快指针：遍历数组
            if (nums[fast] != nums[slow]) {
                slow++;
                nums[slow] = nums[fast]; // 覆盖重复元素
            }
        }
        return slow + 1; // 去重后数组长度
    }

    public static void main(String[] args) {
        int[] nums = {1,1,2,3,3};
        int newLength = removeDuplicates(nums);
        // 输出去重后的数组：[1,2,3]
        System.out.println(Arrays.toString(Arrays.copyOf(nums, newLength)));
    }
}
```

```java
public class ArrayExpansion {
    public static int[] expandArray(int[] original, int newLength) {
        // 新长度必须大于原长度
        if (newLength <= original.length) {
            return original;
        }
        // 创建新数组
        int[] newArray = new int[newLength];
        // 拷贝原数组元素到新数组
        System.arraycopy(original, 0, newArray, 0, original.length);
        return newArray;
    }

    public static void main(String[] args) {
        int[] arr = {1,2,3};
        int[] expandedArr = expandArray(arr, 6);
        // 输出：[1, 2, 3, 0, 0, 0]
        System.out.println(Arrays.toString(expandedArr));
    }
}
```

2. 判断字符串是否为回文（正读反读都一样）
```java
public static boolean isPalindrome(String str) {
    if (str == null || str.isEmpty()) {
        return true;
    }
    // 预处理：转为小写，去除非字母数字字符
    String cleanStr = str.replaceAll("[^a-zA-Z0-9]", "").toLowerCase();
    int left = 0;
    int right = cleanStr.length() - 1;
    while (left < right) {
        if (cleanStr.charAt(left) != cleanStr.charAt(right)) {
            return false;
        }
        left++;
        right--;
    }
    return true;
}
```

3. 字符串匹配（KMP算法，高效查找子串）  
KMP 算法是字符串匹配的经典算法，时间复杂度 O (n+m)（n 为主串长度，m 为模式串长度）：
```
public class KMPAlgorithm {
    // 构建部分匹配表（前缀函数）
    private static int[] buildNext(String pattern) {
        int m = pattern.length();
        int[] next = new int[m];
        int j = 0; // 前缀指针
        for (int i = 1; i < m; i++) {
            // 回退j
            while (j > 0 && pattern.charAt(i) != pattern.charAt(j)) {
                j = next[j - 1];
            }
            if (pattern.charAt(i) == pattern.charAt(j)) {
                j++;
            }
            next[i] = j;
        }
        return next;
    }

    // KMP匹配：返回子串在主串中首次出现的索引，未找到返回-1
    public static int kmpSearch(String text, String pattern) {
        if (pattern.isEmpty()) return 0;
        int[] next = buildNext(pattern);
        int j = 0; // 模式串指针
        for (int i = 0; i < text.length(); i++) {
            // 回退j
            while (j > 0 && text.charAt(i) != pattern.charAt(j)) {
                j = next[j - 1];
            }
            if (text.charAt(i) == pattern.charAt(j)) {
                j++;
            }
            // 匹配完成
            if (j == pattern.length()) {
                return i - j + 1;
            }
        }
        return -1;
    }

    public static void main(String[] args) {
        String text = "ABABCABCABX";
        String pattern = "ABCABX";
        int index = kmpSearch(text, pattern);
        System.out.println("子串首次出现位置：" + index); // 6
    }
}
```

```java
public static String longestCommonPrefix(String[] strs) {
    if (strs == null || strs.length == 0) {
        return "";
    }
    // 以第一个字符串为基准
    String prefix = strs[0];
    for (int i = 1; i < strs.length; i++) {
        // 不断缩短前缀，直到匹配当前字符串
        while (strs[i].indexOf(prefix) != 0) {
            prefix = prefix.substring(0, prefix.length() - 1);
            if (prefix.isEmpty()) {
                return "";
            }
        }
    }
    return prefix;
}
```

## 单链表
```java
// 单链表节点类
class ListNode {
    int val;          // 节点值
    ListNode next;    // 指向下一个节点的引用
    // 构造方法
    ListNode() {}
    ListNode(int val) { this.val = val; }
    ListNode(int val, ListNode next) { this.val = val; this.next = next; }
}
// 单链表的核心操作 增删改查遍历
public class SinglyLinkedList {
    // 头节点（哨兵节点，简化边界处理）
    private ListNode head;
    // 链表长度
    private int size;

    // 构造方法：初始化空链表
    public SinglyLinkedList() {
        head = new ListNode(0); // 虚拟头节点（避免处理头节点为空的情况）
        size = 0;
    }

    // 1. 获取指定索引的节点值（索引从0开始）
    public int get(int index) {
        // 边界检查
        if (index < 0 || index >= size) {
            throw new IndexOutOfBoundsException("索引越界");
        }
        ListNode curr = head.next; // 跳过虚拟头节点
        for (int i = 0; i < index; i++) {
            curr = curr.next;
        }
        return curr.val;
    }

    // 2. 在链表头部添加节点
    public void addAtHead(int val) {
        addAtIndex(0, val);
    }

    // 3. 在链表尾部添加节点
    public void addAtTail(int val) {
        addAtIndex(size, val);
    }

    // 4. 在指定索引插入节点
    public void addAtIndex(int index, int val) {
        // 边界检查：index>size则插入失败；index<0则插入头部
        if (index > size) return;
        if (index < 0) index = 0;
        
        ListNode prev = head;
        // 找到插入位置的前驱节点
        for (int i = 0; i < index; i++) {
            prev = prev.next;
        }
        // 创建新节点并插入
        ListNode newNode = new ListNode(val);
        newNode.next = prev.next;
        prev.next = newNode;
        size++;
    }

    // 5. 删除指定索引的节点
    public void deleteAtIndex(int index) {
        if (index < 0 || index >= size) return;
        
        ListNode prev = head;
        // 找到待删除节点的前驱节点
        for (int i = 0; i < index; i++) {
            prev = prev.next;
        }
        // 删除节点
        prev.next = prev.next.next;
        size--;
    }

    // 6. 遍历链表（打印所有节点值）
    public void traverse() {
        ListNode curr = head.next;
        StringBuilder sb = new StringBuilder();
        while (curr != null) {
            sb.append(curr.val).append(" -> ");
            curr = curr.next;
        }
        sb.append("null");
        System.out.println(sb.toString());
    }

    // 测试
    public static void main(String[] args) {
        SinglyLinkedList list = new SinglyLinkedList();
        list.addAtHead(1);      // 1 -> null
        list.addAtTail(3);      // 1 -> 3 -> null
        list.addAtIndex(1, 2);  // 1 -> 2 -> 3 -> null
        list.traverse();        // 输出：1 -> 2 -> 3 -> null
        System.out.println(list.get(1)); // 输出：2
        list.deleteAtIndex(1);  // 1 -> 3 -> null
        list.traverse();        // 输出：1 -> 3 -> null
    }
}
```

1. 链表反转（迭代法，空间复杂度O(1)）
```java
// 反转单链表，返回新的头节点
public ListNode reverseList(ListNode head) {
    ListNode prev = null;    // 前驱节点
    ListNode curr = head;    // 当前节点
    while (curr != null) {
        ListNode nextTemp = curr.next; // 暂存下一个节点
        curr.next = prev;             // 反转指针
        prev = curr;                  // 前驱节点后移
        curr = nextTemp;              // 当前节点后移
    }
    return prev; // 最终prev是新的头节点
}

// 测试：1->2->3 → 3->2->1
```

2. 检查链表是否有环（快慢指针法）
```java
// 快慢指针：快指针每次走2步，慢指针走1步，相遇则有环
public boolean hasCycle(ListNode head) {
    if (head == null || head.next == null) {
        return false;
    }
    ListNode slow = head;
    ListNode fast = head.next;
    while (slow != fast) {
        if (fast == null || fast.next == null) {
            return false; // 快指针到尾部，无环
        }
        slow = slow.next;
        fast = fast.next.next;
    }
    return true;
}
```

3. 找到两个链表的相交节点
```java
// 双指针法：走完自己的链表走对方的链表，相遇点即为相交节点
public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
    if (headA == null || headB == null) return null;
    ListNode pA = headA, pB = headB;
    while (pA != pB) {
        pA = (pA == null) ? headB : pA.next;
        pB = (pB == null) ? headA : pB.next;
    }
    return pA;
}
```
4. 删除链表的倒数第 N 个节点
```java
// 快慢指针：快指针先走n步，然后快慢指针一起走，快指针到尾部时，慢指针指向倒数第n个节点的前驱
public ListNode removeNthFromEnd(ListNode head, int n) {
    ListNode dummy = new ListNode(0, head); // 虚拟头节点
    ListNode fast = head;
    ListNode slow = dummy;
    
    // 快指针先走n步
    for (int i = 0; i < n; i++) {
        fast = fast.next;
    }
    
    // 快慢指针一起走
    while (fast != null) {
        fast = fast.next;
        slow = slow.next;
    }
    
    // 删除节点
    slow.next = slow.next.next;
    return dummy.next;
}
```
