有序数组 / 字符串的两端向中间逼近的问题（两数之和、反转字符串、回文判断、二分法变种）

```java
/**
 * 相向双指针通用模板
 * @param nums 有序数组（核心前提）
 * @param target 目标值（根据场景调整）
 * @return 符合条件的结果
 */
public int twoPointersOpposite(int[] nums, int target) {
    // 1. 初始化左右指针：左指针在头，右指针在尾
    int left = 0;
    int right = nums.length - 1;
    
    // 2. 循环条件：左指针不超过右指针（避免越界）
    while (left < right) { // 注意：根据场景选 left < right 或 left <= right
        // 3. 计算当前状态（和/差/比较结果）
        int sum = nums[left] + nums[right];
        
        // 4. 根据条件移动指针
        if (sum == target) {
            // 找到目标，返回结果（或记录结果）
            return left; // 示例，根据场景调整
        } else if (sum < target) {
            // 需增大值，左指针右移
            left++;
        } else {
            // 需减小值，右指针左移
            right--;
        }
    }
    
    // 5. 未找到/处理完成，返回默认值
    return -1;
}
```

// LeetCode 167：有序数组中找两个数之和等于target，返回下标（从1开始）
public int[] twoSum(int[] numbers, int target) {
    int left = 0;
    int right = numbers.length - 1;
    while (left < right) {
        int sum = numbers[left] + numbers[right];
        if (sum == target) {
            return new int[]{left + 1, right + 1};
        } else if (sum < target) {
            left++;
        } else {
            right--;
        }
    }
    return new int[]{-1, -1}; // 题目保证有解，此处为兜底
}
