```
# 本地安装（Go 1.16+）
go install github.com/go-delve/delve/cmd/dlv@latest

# 远程服务器安装（执行相同命令，需先安装Go环境）
ssh <远程IP> "go install github.com/go-delve/delve/cmd/dlv@latest"

# 验证安装（本地/远程都执行）
dlv version  # 输出类似：Delve Debugger v1.26.0


go build -gcflags "all=-N -l" -o myapp main.go

# 远程服务器执行（替换为你的代码路径/入口文件）
~/go/bin/dlv debug \
  --headless \          # 无头模式（不启动本地交互终端）
  --listen=0.0.0.0:2345 \ # 监听所有网卡的2345端口（必须0.0.0.0，否则本地连不上）
  --api-version=2 \     # 指定API版本，兼容本地dlv
  --accept-multiclient \ # 允许多个客户端连接（比如同时VSCode+命令行）
  --log \               # 输出调试日志（排查连接问题时开启）
  --init \              # 等待连接
  /path/to/your/main.go # 远程代码的入口文件（或项目根目录）
~/go/bin/dlv debug \
  --headless \
  --listen=0.0.0.0:2345 \
  --api-version=2 \
  --accept-multiclient \
  --log \
  --init \
  ./main.go

# 调试编译后的二进制（推荐，无需保留源码）
~/go/bin/dlv exec \
  --headless \          # 无头模式（无本地终端）
  --listen=0.0.0.0:2345 \ # 监听所有网卡的2345端口
  --api-version=2 \     # 兼容Windows的dlv版本
  --accept-multiclient \ # 允许多客户端连接
  --log \               # 输出调试日志（排查问题用）
  ./myapp               # 要调试的二进制文件
  # 可选：添加程序参数，比如 --config /etc/app.yaml
```
